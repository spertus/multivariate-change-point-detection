multichangepoints workflow

This vignette demonstrates the core workflow for sequential change-point detection with:

  • IID Gaussian data generating process (DGP)
  • simple-vs-simple test supermartingale (TSM)
  • Shiryaev-Roberts (S-R) detector
devtools::load_all("~/Dropbox/CANSSI/multi-change-points/")
#> ℹ Loading multichangepoints
#> Warning: package 'testthat' was built under R version 4.5.2
library(multichangepoints)
set.seed(20260305)

1. Specify a Gaussian model and TSM

# define a simple change model, including all pre-change and post-change parameters
model <- GaussianModel(
  mean_pre = 0,
  sd_pre = 1,
  mean_post = 1,
  sd_post = 1
)

# construct a tsm object from the model, it will receive data
tsm <- TSM(model)

2. Create an IID Gaussian DGP

# set up the data generating process for the data sequence, including the pre/post distributions and the change-point
dgp <- DGP(
  generator = default_gaussian_dgp,
  pre_params = list(mean = 0, sd = 1), # parameters pre-change
  post_params = list(mean = 1, sd = 1), # parameters post-change
  nu = 120, # change-point
  name = "iid-gaussian-change"
)

# generate data from the dgp
N <- 250
x <- generate_stream(dgp, N = N, K = 1)
head(x)
#> [1] -0.2698252 -0.9911324 -0.4595598 -0.8808831 -1.4012064  1.9237771

3. Build increments and run the S-R detector with ARL control

# increment sequence from data x
inc <- compute_increments(tsm, x, log = TRUE)

# instantiate a detector class with ARL control
sr_1 <- ShiryaevRobertsDetector(
  alpha = 0.05, # average run length without false alarm is only 20
  criterion = "ARL"
)
sr_2 <- ShiryaevRobertsDetector(
  alpha = 0.001, # average run length without false alarm is 1000
  criterion = "ARL"
)


# push increments through detector sr
det_out_1 <- run_detector(sr_1, inc, log = TRUE)
det_out_2 <- run_detector(sr_2, inc, log = TRUE)
str(det_out_1)
#> List of 5
#>  $ statistic    : num [1:250] -0.77 -1.111 -0.675 -0.969 -1.58 ...
#>  $ stopping_time: int 14
#>  $ alarm        : logi TRUE
#>  $ criterion    : chr "ARL"
#>  $ log          : logi TRUE

The detector returns:

  • statistic: the S-R path over time
  • stopping_time: first alarm time (or Inf)
  • alarm: whether an alarm occurred by the horizon

The following figure plots the path of the TSM and ALR-controlled S-R statistic, along with the alarm time.

5. Estimate operating characteristics by simulation

sr <- ShiryaevRobertsDetector(
  alpha = 0.001,
  criterion = "ARL"
)

mc <- run_simulation(
  detector = sr,
  tsm = tsm,
  dgp = dgp,
  n_rep = 100,
  N = 300,
  seed = 77
)

mc
#>           detector                 dgp n_rep horizon  nu false_alarm_prob
#> 1 shiryaev-roberts iid-gaussian-change   100     300 120              0.1
#>      ARL   ADD   CAD
#> 1 125.12 10.12 10.12

This returns Monte Carlo summaries including:

  • false_alarm_prob: proportion of runs alarming before (or at) the change
  • ARL: average run length to alarm (censored at horizon)
  • CAD: average detection delay after the change

Multivariate detection

Simple-vs-simple multivariate detection. Note that when the alternative is mis-specified, the helper TSM does not grow and the detector can be very inefficient.

K <- 3 # number of streams
N <- 50 # length of streams
dgp_mvn <- DGP(
  generator = default_multivariate_gaussian_dgp,
  pre_params = list(mu = c(0, 0, 0), Sigma = diag(K)), # parameters pre-change
  post_params = list(mu = c(1, 0, 0), Sigma = diag(K)), # parameters post-change
  nu = 25, # change-point
  name = "mv-gauss-change"
)

# generate data from the dgp
x <- generate_stream(dgp_mvn, N = N, K = K)

# simple vs simple
# true alternative
model_true <- MultivariateGaussianModel(
  mu_pre = c(0, 0, 0),
  Sigma_pre = diag(K),
  mu_post = c(1, 0, 0),
  Sigma_post = diag(K)
)
# mis-specified alternative
model_miss <- MultivariateGaussianModel(
  mu_pre = c(0, 0, 0),
  Sigma_pre = diag(K),
  mu_post = c(1, 1, 1),
  Sigma_post = diag(K)
)
# composite post-change model
# estimating the post-change mean
model_comp_pred <- MultivariateGaussianModel(
  mu_pre = c(0, 0, 0),
  Sigma_pre = diag(K),
  mu_post = rbind(c(0,1), c(0,1), c(0,1)),
  Sigma_post = diag(K),
  method = "predictable"
)
# mixing over the post-change mean
model_comp_mix <- MultivariateGaussianModel(
  mu_pre = c(0, 0, 0),
  Sigma_pre = diag(K),
  mu_post = rbind(c(0,1), c(0,1), c(0,1)),
  Sigma_post = diag(K),
  method = "mixture"
)

# instantiate TSMs and compute increments
inc_true <- compute_increments(TSM(model_true), x, log = TRUE)
inc_miss <- compute_increments(TSM(model_miss), x, log = TRUE)
inc_comp_pred <- compute_increments(TSM(model_comp_pred), x, log = TRUE)
inc_comp_mix <- compute_increments(TSM(model_comp_mix), x, log = TRUE)

# instantiate detector and compute sequence
sr <-  ShiryaevRobertsDetector(alpha = 0.001, criterion = "ARL")
det_out_true <- run_detector(sr, inc_true, log = TRUE)
det_out_miss <- run_detector(sr, inc_miss, log = TRUE)
det_out_comp_pred <- run_detector(sr, inc_comp_pred, log = TRUE)
det_out_comp_mix <- run_detector(sr, inc_comp_mix, log = TRUE)
old_par <- par(mfrow = c(1, 3))

# Panel 1: K data streams
matplot(x, type = "l", lty = 1, lwd = 1.5,
        xlab = "time", ylab = "x_tk", main = "Data Streams", ylim = c(-5, 5))
abline(v = dgp_mvn@nu, lty = 2, col = "gray40")
legend("topleft",
       legend = paste0("stream ", seq_len(ncol(x))),
       col = seq_len(ncol(x)), lty = 1, lwd = 1.5, cex = 0.8, bty = "n")

# Panel 2: helper TSMs from increments
tsm_true <- compute_tsm(TSM(model_true), x, log = TRUE)
tsm_miss <- compute_tsm(TSM(model_miss), x, log = TRUE)
tsm_comp_pred <- increments_to_tsm(inc_comp_pred, log = TRUE)
tsm_comp_mix <- increments_to_tsm(inc_comp_mix, log = TRUE)
plot(tsm_true, type = "l", lwd = 2, col = "darkgreen",
     xlab = "time", ylab = "log helper TSM", main = "Helper TSMs", ylim = range(c(tsm_true, tsm_miss)))
lines(tsm_miss, lwd = 2, col = "darkorange")
lines(tsm_comp_pred, lwd = 2, col = "steelblue")
lines(tsm_comp_mix, lwd = 2, col = "firebrick")
abline(v = dgp_mvn@nu, lty = 2, col = "gray40")
legend("topleft",
       legend = c("true model", "mis-specified model", "predictable", "mixture"),
       col = c("darkgreen", "darkorange", "steelblue","firebrick"),
       lty = 1, lwd = 2, cex = 0.8, bty = "n")

# Panel 3: detector statistics
plot(det_out_true$statistic, type = "l", lwd = 2, col = "darkgreen",
     xlab = "time", ylab = "log R_t", main = "S-R Statistics")
lines(det_out_miss$statistic, lwd = 2, col = "darkorange")
lines(det_out_comp_pred$statistic, lwd = 2, col = "steelblue")
lines(det_out_comp_mix$statistic, lwd = 2, col = "firebrick")
abline(h = log(sr@threshold), lty = 3, col = "gray40")
abline(v = dgp_mvn@nu, lty = 2, col = "gray40")
if (is.finite(det_out_true$stopping_time)) {
  abline(v = det_out_true$stopping_time, lwd = 1.5, col = "darkgreen")
}
if (is.finite(det_out_miss$stopping_time)) {
  abline(v = det_out_miss$stopping_time, lwd = 1.5, col = "darkorange")
}
if (is.finite(det_out_comp_pred$stopping_time)) {
  abline(v = det_out_comp_pred$stopping_time, lwd = 1.5, col = "steelblue")
}
if (is.finite(det_out_comp_mix$stopping_time)) {
  abline(v = det_out_comp_mix$stopping_time, lwd = 1.5, col = "firebrick")
}
legend("topleft",
       legend = c("true model", "mis-specified model", "predictable", "mixture"),
       col = c("darkgreen", "darkorange", "steelblue", "firebrick"),
       lty = 1, lwd = 2, cex = 0.8, bty = "n")

Combining

The same x from above has K = 3 independent Gaussian streams with a sparse change: only stream 1 shifts from mean 0 to mean 1. We fit a marginal simple-vs-simple TSM for each stream using the true marginal parameters, then combine the resulting increment sequences three ways: product (assumes independence), average (valid under arbitrary dependence), and universal portfolio (valid under arbitrary dependence, adapts weights online).

# ---- marginal models, one per stream, from the MVN DGP parameters ----
mu_pre  <- dgp_mvn@pre_params$mu
mu_post <- dgp_mvn@post_params$mu
sd_pre  <- sqrt(diag(dgp_mvn@pre_params$Sigma))
sd_post <- sqrt(diag(dgp_mvn@post_params$Sigma))

marg_tsms <- lapply(seq_len(K), function(k) {
  TSM(GaussianModel(
    mean_pre  = mu_pre[k],
    sd_pre    = sd_pre[k],
    mean_post = mu_post[k],
    sd_post   = sd_post[k]
  ))
})

# marginal log-increments: N x K matrix (one column per stream)
marg_inc <- sapply(seq_len(K), function(k) {
  compute_increments(marg_tsms[[k]], x[, k], log = TRUE)
})

# ---- combine marginal increments ----
inc_avg <- combine_streams(AverageCombiner(),            marg_inc, log = TRUE)
inc_prd <- combine_streams(ProductCombiner(),            marg_inc, log = TRUE)
inc_up  <- combine_streams(UniversalPortfolioCombiner(), marg_inc, log = TRUE)

# ---- run the same detector on each combined sequence ----
sr_comb <- ShiryaevRobertsDetector(alpha = 0.001, criterion = "ARL")
det_avg <- run_detector(sr_comb, inc_avg, log = TRUE)
det_prd <- run_detector(sr_comb, inc_prd, log = TRUE)
det_up  <- run_detector(sr_comb, inc_up,  log = TRUE)
old_par <- par(mfrow = c(1, 3))

# Panel 1: data streams (same data as above)
matplot(x, type = "l", lty = 1, lwd = 1.5,
        xlab = "time", ylab = "x_tk", main = "Data Streams", ylim = c(-3, 3))
abline(v = dgp_mvn@nu, lty = 2, col = "gray40")
legend("topleft",
       legend = paste0("stream ", seq_len(K)),
       col = seq_len(K), lty = 1, lwd = 1.5, cex = 0.8, bty = "n")

# Panel 2: combined TSM paths
tsm_avg <- increments_to_tsm(inc_avg, log = TRUE)
tsm_prd <- increments_to_tsm(inc_prd, log = TRUE)
tsm_up  <- increments_to_tsm(inc_up,  log = TRUE)

ylim_tsm <- range(c(tsm_avg, tsm_prd, tsm_up))
plot(tsm_prd, type = "l", lwd = 2, col = "darkgreen",
     xlab = "time", ylab = "log combined TSM", main = "Combined TSMs",
     ylim = ylim_tsm)
lines(tsm_up,  lwd = 2, col = "firebrick")
lines(tsm_avg, lwd = 2, col = "steelblue")
abline(v = dgp_mvn@nu, lty = 2, col = "gray40")
legend("topleft",
       legend = c("product", "universal portfolio", "average"),
       col    = c("darkgreen", "firebrick", "steelblue"),
       lty = 1, lwd = 2, cex = 0.8, bty = "n")

# Panel 3: S-R statistics
ylim_sr <- range(c(det_avg$statistic, det_prd$statistic, det_up$statistic))
plot(det_prd$statistic, type = "l", lwd = 2, col = "darkgreen",
     xlab = "time", ylab = "log R_t", main = "S-R Statistics",
     ylim = ylim_sr)
lines(det_up$statistic,  lwd = 2, col = "firebrick")
lines(det_avg$statistic, lwd = 2, col = "steelblue")
abline(h = log(sr_comb@threshold), lty = 3, col = "gray40")
abline(v = dgp_mvn@nu, lty = 2, col = "gray40")
for (res in list(list(d = det_prd, col = "darkgreen"),
                 list(d = det_up,  col = "firebrick"),
                 list(d = det_avg, col = "steelblue"))) {
  if (is.finite(res$d$stopping_time)) {
    abline(v = res$d$stopping_time, lwd = 1.5, col = res$col)
  }
}
legend("topleft",
       legend = c("product", "universal portfolio", "average"),
       col    = c("darkgreen", "firebrick", "steelblue"),
       lty = 1, lwd = 2, cex = 0.8, bty = "n")
Combining marginal TSMs under sparse change (stream 1 only). Product combining matches the oracle; universal portfolio adapts toward it; average combining is roughly K-times weaker.
